    <pre>
    &lt;html&gt;  
      &lt;head&gt;  
        <b>&lt;style&gt; 
          h1 {color: red;}
        &lt;/style&gt; </b>
      &lt;/head&gt;
      
      &lt;body&gt;  
        <b>&lt;h1&gt; A heading &lt;/h1&gt; </b>
      &lt;/body&gt;
    &lt;/html&gt;
    </pre>
  </li>
</ul>
